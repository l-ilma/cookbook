generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}


model User {
  id        Int      @id @default(autoincrement())
  username  String   @db.VarChar(15)
  password  String   @db.VarChar(15)
  title     String   @db.VarChar(255)
  profileImagePath String @db.VarChar(255)
  email   String   @unique
  recipes Recipe []
  reviews Review []
  userFavorites UserFavorite []
}


model Recipe {
  id      Int      @id @default(autoincrement())
  userId  Int
  user    User     @relation(fields: [userId], references: [id])
  description    String @db.VarChar(2048)
  imagesPath String @db.VarChar(255)
  price Float
  categoryId Int
  category Category @relation(fields: [categoryId], references: [id])
  createdAt DateTime 
  reviews Review []
  userFavorites UserFavorite []
}


model Category {
  id Int @id @default(autoincrement())
  name String @db.VarChar(255)
  recipes Recipe []
}


model Review {
  id Int @id @default(autoincrement())
  recipeId Int
  recipe Recipe @relation(fields: [recipeId], references: [id])
  userId Int
  user User @relation(fields: [userId], references: [id])
  text String @db.VarChar(1024)
  evaluation Float
  createdAt DateTime  
}


model UserFavorite {
  recipeId Int
  recipe Recipe @relation(fields: [recipeId], references: [id])
  userId Int
  user User @relation(fields: [userId], references: [id])
  @@id([recipeId, userId])
}
